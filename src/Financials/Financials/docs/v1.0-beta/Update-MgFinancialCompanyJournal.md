 [-AccountCategory <String>] [-AccountDisplayName <String>] [-AccountId <String>]
 [-AccountLastModifiedDateTime <DateTime>] [-AccountNumber <String>] [-AccountSubCategory <String>]
 [-BalancingAccountId <String>] [-BalancingAccountNumber <String>] [-Code <String>] [-DisplayName <String>]
 [-Id <String>] [-JournalLines <IMicrosoftGraphJournalLine[]>] [-LastModifiedDateTime <DateTime>] [-PassThru]
 [-Confirm] [-WhatIf] [<CommonParameters>]
```
