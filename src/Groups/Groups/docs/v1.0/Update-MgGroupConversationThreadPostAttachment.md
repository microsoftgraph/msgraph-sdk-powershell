Update-MgGroupConversationThreadPostAttachment -AttachmentId <String> -ConversationId <String>
 -ConversationThreadId <String> -GroupId <String> -PostId <String> [-AdditionalProperties <Hashtable>]
 [-ContentType <String>] [-Id <String>] [-IsInline] [-LastModifiedDateTime <DateTime>] [-Name <String>]
 [-Size <Int32>] [-PassThru] [-Confirm] [-WhatIf] [<CommonParameters>]
